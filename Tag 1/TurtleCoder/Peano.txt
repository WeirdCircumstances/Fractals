function main() {
	turtle->setSpeed(10000)
	var step := 100
	var iteration := 3
	var step := myStepLengh(step,iteration)
	for(var i < 2) {
		turtle->penUp()
		turtle->forward(myGoToStart(step,iteration))
		turtle->leftTurn(90)
		turtle->penDown()
	}
	turtle->leftTurn(90)
	myPeanoX(step,iteration)
}
function myPeanoX(step,iteration) {
	if(iteration > 0) {
	    myPeanoX(step,iteration-1)
	    turtle->forward(step)
		myPeanoY(step,iteration-1)
		turtle->forward(step)
		myPeanoX(step,iteration-1)
		turtle->rightTurn(90)
		turtle->forward(step)
		turtle->rightTurn(90)
		myPeanoY(step,iteration-1)
		turtle->forward(step)
		myPeanoX(step,iteration-1)
		turtle->forward(step)
		myPeanoY(step,iteration-1)
		turtle->leftTurn(90)
		turtle->forward(step)
		turtle->leftTurn(90)
		myPeanoX(step,iteration-1)
	    turtle->forward(step)
		myPeanoY(step,iteration-1)
		turtle->forward(step)
		myPeanoX(step,iteration-1)
	}else {
		return 0
	}

}
function myPeanoY(step,iteration) {
	if(iteration > 0) {
	    myPeanoY(step,iteration-1)
		turtle->forward(step)
		myPeanoX(step,iteration-1)
		turtle->forward(step)
		myPeanoY(step,iteration-1)
		turtle->leftTurn(90)
		turtle->forward(step)
		turtle->leftTurn(90)
		myPeanoX(step,iteration-1)
	    turtle->forward(step)
		myPeanoY(step,iteration-1)
		turtle->forward(step)
		myPeanoX(step,iteration-1)
		turtle->rightTurn(90)
		turtle->forward(step)
		turtle->rightTurn(90)
		myPeanoY(step,iteration-1)
		turtle->forward(step)
		myPeanoX(step,iteration-1)
		turtle->forward(step)
		myPeanoY(step,iteration-1)
	}else {
		return 0
	}

}
function myStepLengh(step,iteration) {
	while(iteration > 0) {
		var step := step / 2
		var iteration := iteration - 1
	}
	return step
}
function myGoToStart(step,iteration) {
	var iteration := iteration * iteration * iteration
	var iteration := iteration / 2
	var step := step * iteration
	return step
}
