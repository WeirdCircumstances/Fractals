function myPfeilR(step,iteration) {
	if(iteration > 0) {
		turtle->leftTurn(60)
		myPfeilL(step,iteration-1)
		turtle->rightTurn(60)
		myPfeilR(step,iteration-1)
		turtle->rightTurn(60)
		myPfeilL(step,iteration-1)
		turtle->leftTurn(60)
	}else {
		turtle->forward(step)
	}

}
function myPfeilL(step,iteration) {
	if(iteration > 0) {
		turtle->rightTurn(60)
		myPfeilR(step,iteration-1)
		turtle->leftTurn(60)
		myPfeilL(step,iteration-1)
		turtle->leftTurn(60)
		myPfeilR(step,iteration-1)
		turtle->rightTurn(60)
	}else {
		turtle->forward(step)
	}

}
